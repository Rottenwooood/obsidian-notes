```
static int find_main_op(int p,int q){
  int priority = 100;
  int deepth = 0;
  int op = -1;

  for(int i = q;i >= p;i--){
    if(tokens[i].type == TK_RP) deepth++;
    else if(tokens[i].type == TK_LP) deepth--;
    if(deepth > 0) continue;

    int pri;
    switch(tokens[i].type){
      case TK_PLUS:
      case TK_MINUS: pri = 1;break;
      case TK_DIV:
      case TK_STAR: pri = 2;break;
      default: continue;
    }
    if(pri <= priority){
      priority = pri;
      op = i;
    }
  }
  return op;
}
```
