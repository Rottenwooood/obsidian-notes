关于Makefile：
```
# 编译器
CC = gcc

# 编译选项
CFLAGS = -Wall -g

# 目标文件
TARGET = helloworld.exe

# 源文件
SOURCES = main.c

# 生成目标
$(TARGET): $(SOURCES)
    $(CC) $(CFLAGS) -o $(TARGET) $(SOURCES)

# 清理生成文件
clean:
    rm -f $(TARGET)

# 伪目标
.PHONY: clean
```
	注意：缩进必须用Tab
	如果编译c++程序，则选择g++
